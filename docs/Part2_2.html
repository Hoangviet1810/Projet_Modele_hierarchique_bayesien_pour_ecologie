1 PROJET STATISTIQUES COMPUTATIONNELLES

Algorithme HMC - Modèle Hiérarchique Poisson-Normal

Hyperparamètres : \(\sigma^2 \sim \text{Exp}(1)\), \(\mu \sim \mathcal{N}(0,4)\)

Reparamétrisation : \(\gamma = \log(\sigma^2)\), \(Z_i = \log(\lambda_i)\)

Il y a 5 fonction. Ce sont:

  • fonction log_posterior_target : calculer la log posterior
  • fonction gradient_target : calculer les gradients de log_postierior par les parametres \(Z_i\), \(\mu\), \(\gamma\)
  • fonction leapfrog : calculer les steps pour algo HMC
  • fonction hmc_sampler : algo HMC
  • fonction run_hmc_chains_sequentially : pour faire des chaines en même temps

1.3 DÉFINITION DES FONCTIONS MATHÉMATIQUES

Définitions des fonctions : - Log à postériori - dérivées partielles - gradient

# Calcul de la Log-Postériori (L)
log_posterior_target <- function(theta, J_list, sum_y_list, I) {
    alphas <- theta[1:I]
    mu     <- theta[I + 1]
    gamma  <- theta[I + 2]
    
    # Terme 1: Vraisemblance de Poisson
    log_lik <- sum(-J_list * exp(alphas) + alphas * sum_y_list)
    
    # Terme 2: Loi a priori conditionnelle des alphas (Loi Normale)
    log_prior_alphas <- -0.5 * exp(-gamma) * sum((alphas - mu)^2) - (I/2) * gamma
    
    # Terme 3: Hyperpriors (mu ~ N(0, 4) et tau^2 ~ Exp(1) reparamétré en gamma)
    log_hyperpriors <- -(mu^2) / 8 - exp(gamma) + gamma
    
    return(log_lik + log_prior_alphas + log_hyperpriors)
}

# Calcul du Gradient de l'Énergie Hamiltonienne (nabla H)
# Rappel : H = -L, donc on implémente les formules dérivées trouvées précédemment
gradient_target <- function(theta, J_list, sum_y_list, I) {
    alphas <- theta[1:I]
    mu     <- theta[I + 1]
    gamma  <- theta[I + 2]
    
    grads <- numeric(I + 2)
    
    # Dérivées partielles par rapport aux alphas (Vectorisé)
    grads[1:I] <- J_list * exp(alphas) - sum_y_list + (alphas - mu) / exp(gamma)
    
    # Dérivée partielle par rapport à mu
    grads[I + 1] <- (mu / 4) - sum(alphas - mu) / exp(gamma)
    
    # Dérivée partielle par rapport à gamma
    grads[I + 2] <- (I / 2 - 1) - 0.5 * exp(-gamma) * sum((alphas - mu)^2) + exp(gamma)
    
    return(grads)
}

1.4 Implémentation des algorithmes

# Algorithme Leapfrog (Saute-Mouton)
leapfrog <- function(q, p, step_size, L_steps, J_list, sum_y_list, I) {
    # Demi-pas initial pour l'impulsion (Attention : signe MOINS car on minimise H)
    p <- p - 0.5 * step_size * gradient_target(q, J_list, sum_y_list, I)
    
    for (l in 1:L_steps) {
        # Pas complet pour la position
        q <- q + step_size * p
        
        # Pas complet pour l'impulsion (sauf au dernier tour)
        if (l < L_steps) {
            p <- p - step_size * gradient_target(q, J_list, sum_y_list, I)
        }
    }
    
    # Demi-pas final pour l'impulsion
    p <- p - 0.5 * step_size * gradient_target(q, J_list, sum_y_list, I)
    
    # Inversion de l'impulsion (pour symétrie de la chaîne de Markov)
    p <- -p 
    
    return(list(q = q, p = p))
}

# Fonction principale HMC
hmc_sampler <- function(init_theta, iterations, step_size, L_steps, J_list, sum_y_list, I) {
    d <- length(init_theta)
    samples <- matrix(NA, nrow = iterations, ncol = d)
    samples[1, ] <- init_theta
    
    acceptances <- 0
    
    for (t in 2:iterations) {
        current_q <- samples[t - 1, ]
        
        # 1. Tirage de l'impulsion aléatoire
        current_p <- rnorm(d, mean = 0, sd = 1)
        
        # 2. Simulation de la trajectoire
        traj <- leapfrog(current_q, current_p, step_size, L_steps, J_list, sum_y_list, I)
        proposed_q <- traj$q
        proposed_p <- traj$p
        
        # 3. Évaluation de l'Énergie Hamiltonienne (H = -L + K)
        current_U <- -log_posterior_target(current_q, J_list, sum_y_list, I)
        current_K <- sum(current_p^2) / 2
        current_H <- current_U + current_K
        
        proposed_U <- -log_posterior_target(proposed_q, J_list, sum_y_list, I)
        proposed_K <- sum(proposed_p^2) / 2
        proposed_H <- proposed_U + proposed_K
        
        # === CRASH TEST DIAGNOSTIC (intégré et silencieux si tout va bien) ===
        if (is.na(proposed_H) || proposed_H == Inf) {
            cat(sprintf("\n[ALERTE] Itération %d: Trajectoire divergente.\n", t))
            cat(sprintf("Gamma: %.3f | Max Alpha: %.3f\n", proposed_q[I+2], max(proposed_q[1:I])))
        }
        
        # 4. Étape de Metropolis
        # Probabilité d'acceptation : min(1, exp(current_H - proposed_H))
        log_acceptance <- current_H - proposed_H
        
        # Protection contre les NA/NaN
        if (is.na(log_acceptance)) log_acceptance <- -Inf 
        
        if (log(runif(1)) < log_acceptance) {
            samples[t, ] <- proposed_q
            acceptances <- acceptances + 1
        } else {
            samples[t, ] <- current_q
        }
        
        # Affichage de la progression tous les 10%
        if (t %% max(1, floor(iterations / 10)) == 0) {
            cat(sprintf("Progression : %d / %d itérations...\n", t, iterations))
        }
    }
    
    cat(sprintf("\nTaux d'acceptation final : %.1f %%\n", (acceptances / (iterations - 1)) * 100))
    return(samples)
}

1.5 EXÉCUTION DU MODÈLE (Wrapper multi-chaînes)

run_hmc_chains_sequentially <- function(n_chains, iterations, step_size, L_steps) {
    all_chains <- list()
    for (chain in 1:n_chains) {
        cat(sprintf("\n--- DÉMARRAGE DE LA CHAÎNE %d ---\n", chain))
        
        # Initialisation sur-dispersée intelligente
        init_mu <- rnorm(1, mean = 0, sd = 2)
        # On tire gamma directement dans N(0,1) pour bien explorer
        init_gamma <- rnorm(1, mean = 0, sd = 1) 
        init_alphas <- rnorm(I, mean = init_mu, sd = sqrt(exp(init_gamma)))
        
        theta_0 <- c(init_alphas, init_mu, init_gamma)
        
        # Lancement
        samples <- hmc_sampler(
            init_theta = theta_0, 
            iterations = iterations, 
            step_size = step_size, 
            L_steps = L_steps, 
            J_list = J_list, 
            sum_y_list = sum_y_list, 
            I = I
        )
        
        # POST-TRAITEMENT : On remet tau^2 et lambda à l'endroit pour l'utilisateur
        # Colonnes 1 à I : alphas -> lambdas (exponentielle)
        samples[, 1:I] <- exp(samples[, 1:I])
        # Colonne I+2 : gamma -> tau^2 (exponentielle)
        samples[, I+2] <- exp(samples[, I+2])
        
        # Optionnel: renommer les colonnes pour que ce soit propre
        colnames(samples) <- c(paste0("lambda_", 1:I), "mu", "tau2")
        
        all_chains[[chain]] <- samples
    }
    return(all_chains)
}

# 1. Préparation des vraies données
# On regroupe toutes les vraies observations (issues de y_list) en un seul vecteur
y_obs <- unlist(y_list)

# On choisit deux statistiques de test pertinentes pour des données de comptage :
# - La proportion de zéros (les sites sans moustiques)
# - La valeur maximale (les invasions extrêmes)
stat_obs_zero <- mean(y_obs == 0) 
stat_obs_max  <- max(y_obs)       

# 2. Préparation des simulations
# Pour garder un temps de calcul rapide, on va tirer au hasard 1000 échantillons 
# de ta distribution a posteriori (largement suffisant pour le diagnostic)
n_samples_ppc <- min(1000, nrow(samples_df))
set.seed(42) # Reproductibilité
indices_ppc <- sample(1:nrow(samples_df), n_samples_ppc)

rep_stat_zero <- numeric(n_samples_ppc)
rep_stat_max  <- numeric(n_samples_ppc)

cat("Génération des données prédictives a posteriori en cours...\n")
Génération des données prédictives a posteriori en cours...
# 3. Boucle de simulation
for (t in 1:n_samples_ppc) {
    idx <- indices_ppc[t]
    y_rep_t <- numeric(0)
    
    # Pour chaque année i, on simule de nouvelles données avec le lambda_i estimé
    for (i in 1:I) {
        # Extraction du lambda estimé à cette itération MCMC
        lambda_i_t <- samples_df[idx, paste0("lambda_", i)]
        
        # Simulation de J_i (nombre de sites) observations selon la loi de Poisson
        y_rep_t <- c(y_rep_t, rpois(J_list[i], lambda = lambda_i_t))
    }
    
    # On calcule les statistiques sur ce faux jeu de données
    rep_stat_zero[t] <- mean(y_rep_t == 0)
    rep_stat_max[t]  <- max(y_rep_t)
}

# 4. Calcul des "p-values bayésiennes"
# (Probabilité que la statistique simulée soit supérieure ou égale à la vraie)
# Un bon modèle a une p-value bayésienne proche de 0.5 (la vérité est au milieu)
# Si c'est proche de 0 ou 1, le modèle échoue à prédire cet aspect.
p_val_zero <- mean(rep_stat_zero >= stat_obs_zero)
p_val_max  <- mean(rep_stat_max >= stat_obs_max)
